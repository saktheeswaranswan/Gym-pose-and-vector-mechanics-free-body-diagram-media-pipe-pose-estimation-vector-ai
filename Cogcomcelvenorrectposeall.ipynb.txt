        "# ---------------------------\n",
        "# INSTALL (Colab / Jupyter)\n",
        "# ---------------------------\n",
        "!pip install ultralytics opencv-python numpy -q\n",
        "\n",
        "import cv2\n",
        "import numpy as np\n",
        "import math\n",
        "import json\n",
        "from ultralytics import YOLO\n",
        "\n",
        "# ---------------------------\n",
        "# CONFIG\n",
        "# ---------------------------\n",
        "INPUT_VIDEO = \"video.mp4\"\n",
        "OUTPUT_VIDEO = \"inference_video.mp4\"\n",
        "OUTPUT_JSON = \"timeline.json\"\n",
        "MODEL_PATH = \"yolo11n-pose.pt\"\n",
        "MIN_CONF = 0.25\n",
        "\n",
        "# COCO keypoint indices\n",
        "KP = {\n",
        "    \"nose\": 0, \"l_eye\": 1, \"r_eye\": 2, \"l_ear\": 3, \"r_ear\": 4,\n",
        "    \"l_sh\": 5, \"r_sh\": 6, \"l_el\": 7, \"r_el\": 8, \"l_wr\": 9, \"r_wr\": 10,\n",
        "    \"l_hp\": 11, \"r_hp\": 12, \"l_kn\": 13, \"r_kn\": 14, \"l_an\": 15, \"r_an\": 16\n",
        "}\n",
        "\n",
        "# Skeleton connections for COCO-17 pose\n",
        "SKELETON = [\n",
        "    (5, 6), (5, 7), (7, 9), (6, 8), (8, 10),\n",
        "    (5, 11), (6, 12), (11, 12), (11, 13), (13, 15),\n",
        "    (12, 14), (14, 16)\n",
        "]\n",
        "\n",
        "# ---------------------------\n",
        "# MATH HELPERS\n",
        "# ---------------------------\n",
        "class Vec:\n",
        "    @staticmethod\n",
        "    def unit(v):\n",
        "        v = np.asarray(v, dtype=np.float32)\n",
        "        n = np.linalg.norm(v)\n",
        "        return v / n if n > 1e-6 else np.zeros_like(v)\n",
        "\n",
        "    @staticmethod\n",
        "    def perp(v):\n",
        "        v = np.asarray(v, dtype=np.float32)\n",
        "        return np.array([-v[1], v[0]], dtype=np.float32)\n",
        "\n",
        "    @staticmethod\n",
        "    def angle(a, b, c):\n",
        "        a = np.asarray(a, dtype=np.float32)\n",
        "        b = np.asarray(b, dtype=np.float32)\n",
        "        c = np.asarray(c, dtype=np.float32)\n",
        "        ba = Vec.unit(a - b)\n",
        "        bc = Vec.unit(c - b)\n",
        "        return float(np.degrees(np.arccos(np.clip(np.dot(ba, bc), -1.0, 1.0))))\n",
        "\n",
        "# ---------------------------\n",
        "# DRAW HELPERS\n",
        "# ---------------------------\n",
        "class Drawer:\n",
        "    def point(self, frame, p, color=(0, 255, 255), r=4):\n",
        "        if p is None:\n",
        "            return\n",
        "        cv2.circle(frame, tuple(np.round(p).astype(int)), r, color, -1)\n",
        "\n",
        "    def line(self, frame, p1, p2, color=(200, 200, 200), thickness=2):\n",
        "        if p1 is None or p2 is None:\n",
        "            return\n",
        "        cv2.line(frame, tuple(np.round(p1).astype(int)), tuple(np.round(p2).astype(int)), color, thickness)\n",
        "\n",
        "    def arrow(self, frame, p, v, color=(255, 0, 0), txt=None, scale=1.0):\n",
        "        if p is None or v is None:\n",
        "            return\n",
        "        p = np.asarray(p, dtype=np.float32)\n",
        "        v = np.asarray(v, dtype=np.float32) * float(scale)\n",
        "        end = p + v\n",
        "        cv2.arrowedLine(\n",
        "            frame,\n",
        "            tuple(np.round(p).astype(int)),\n",
        "            tuple(np.round(end).astype(int)),\n",
        "            color,\n",
        "            3,\n",
        "            tipLength=0.3\n",
        "        )\n",
        "        if txt is not None:\n",
        "            cv2.putText(\n",
        "                frame,\n",
        "                txt,\n",
        "                tuple(np.round(end).astype(int)),\n",
        "                cv2.FONT_HERSHEY_SIMPLEX,\n",
        "                0.5,\n",
        "                color,\n",
        "                1,\n",
        "                cv2.LINE_AA\n",
        "            )\n",
        "\n",
        "    def arc(self, frame, a, b, c, color=(0, 255, 255)):\n",
        "        if a is None or b is None or c is None:\n",
        "            return\n",
        "        ang1 = math.atan2(a[1] - b[1], a[0] - b[0])\n",
        "        ang2 = math.atan2(c[1] - b[1], c[0] - b[0])\n",
        "        diff = (ang2 - ang1 + math.pi) % (2 * math.pi) - math.pi\n",
        "\n",
        "        pts = []\n",
        "        for t in np.linspace(0, 1, 20):\n",
        "            ang = ang1 + diff * t\n",
        "            x = int(b[0] + 30 * math.cos(ang))\n",
        "            y = int(b[1] + 30 * math.sin(ang))\n",
        "            pts.append([x, y])\n",
        "\n",
        "        pts = np.array(pts, np.int32).reshape(-1, 1, 2)\n",
        "        cv2.polylines(frame, [pts], False, color, 2)\n",
        "\n",
        "# ---------------------------\n",
        "# FORCE / BALANCE / STABILITY\n",
        "# ---------------------------\n",
        "class Force:\n",
        "    def compute(self, knee, ankle, hip, com):\n",
        "        knee = np.asarray(knee, dtype=np.float32)\n",
        "        ankle = np.asarray(ankle, dtype=np.float32)\n",
        "        hip = np.asarray(hip, dtype=np.float32)\n",
        "        com = np.asarray(com, dtype=np.float32)\n",
        "\n",
        "        shank = ankle - knee\n",
        "        normal = Vec.perp(Vec.unit(shank))\n",
        "        knee_angle = Vec.angle(hip, knee, ankle)\n",
        "        kf = np.cos(np.radians(knee_angle))\n",
        "\n",
        "        offset = Vec.unit(com - ankle)\n",
        "        of = float(np.dot(offset, normal))\n",
        "\n",
        "        mag = 100 + 120 * abs(kf) + 80 * abs(of)\n",
        "        vertical = np.array([0, -1], dtype=np.float32)\n",
        "        direction = Vec.unit(0.6 * normal + 0.4 * vertical)\n",
        "        return direction * mag, float(mag)\n",
        "\n",
        "class Stability:\n",
        "    def compute(self, com, la, ra, width):\n",
        "        if com is None or (la is None and ra is None):\n",
        "            return False, 0.0, None\n",
        "\n",
        "        feet = [p for p in (la, ra) if p is not None]\n",
        "        xs = [float(p[0]) for p in feet]\n",
        "        xmin, xmax = min(xs), max(xs)\n",
        "\n",
        "        # Expand support region a bit\n",
        "        margin = 0.12 * width\n",
        "        xmin -= margin\n",
        "        xmax += margin\n",
        "\n",
        "        dist = 0.0\n",
        "        if com[0] < xmin:\n",
        "            dist = xmin - com[0]\n",
        "        elif com[0] > xmax:\n",
        "            dist = com[0] - xmax\n",
        "\n",
        "        score = float(np.exp(-dist / 40.0))\n",
        "        stable = score > 0.5\n",
        "\n",
        "        support_center = np.array([(xmin + xmax) / 2.0, float(np.mean([la[1], ra[1]]) if la is not None and ra is not None else feet[0][1])], dtype=np.float32)\n",
        "        return stable, score, support_center\n",
        "\n",
        "# ---------------------------\n",
        "# MAIN ENGINE\n",
        "# ---------------------------\n",
        "class Engine:\n",
        "    def __init__(self):\n",
        "        self.model = YOLO(MODEL_PATH)\n",
        "        self.draw = Drawer()\n",
        "        self.force = Force()\n",
        "        self.stab = Stability()\n",
        "\n",
        "    def _get(self, kpts, conf, i):\n",
        "        return kpts[i] if conf[i] > MIN_CONF else None\n",
        "\n",
        "    def process_person(self, frame, kpts, conf, person_id):\n",
        "        def get(i):\n",
        "            return self._get(kpts, conf, i)\n",
        "\n",
        "        # Draw skeleton\n",
        "        for a, b in SKELETON:\n",
        "            if conf[a] > MIN_CONF and conf[b] > MIN_CONF:\n",
        "                self.draw.line(frame, kpts[a], kpts[b])\n",
        "\n",
        "        # Draw keypoints\n",
        "        for i in range(len(kpts)):\n",
        "            if conf[i] > MIN_CONF:\n",
        "                self.draw.point(frame, kpts[i])\n",
        "\n",
        "        # Joint angles\n",
        "        joints = [\n",
        "            (KP[\"l_sh\"], KP[\"l_el\"], KP[\"l_wr\"]),\n",
        "            (KP[\"r_sh\"], KP[\"r_el\"], KP[\"r_wr\"]),\n",
        "            (KP[\"l_hp\"], KP[\"l_kn\"], KP[\"l_an\"]),\n",
        "            (KP[\"r_hp\"], KP[\"r_kn\"], KP[\"r_an\"])\n",
        "        ]\n",
        "\n",
        "        angles = {}\n",
        "        for a, b, c in joints:\n",
        "            key = f\"{a}_{b}_{c}\"\n",
        "            if conf[a] > MIN_CONF and conf[b] > MIN_CONF and conf[c] > MIN_CONF:\n",
        "                ang = Vec.angle(kpts[a], kpts[b], kpts[c])\n",
        "                angles[key] = ang\n",
        "                self.draw.arc(frame, kpts[a], kpts[b], kpts[c])\n",
        "                cv2.putText(\n",
        "                    frame,\n",
        "                    f\"{int(round(ang))}°\",\n",
        "                    tuple(np.round(kpts[b]).astype(int)),\n",
        "                    cv2.FONT_HERSHEY_SIMPLEX,\n",
        "                    0.5,\n",
        "                    (255, 255, 255),\n",
        "                    1,\n",
        "                    cv2.LINE_AA\n",
        "                )\n",
        "\n",
        "        # COM / CoG approximation from hips + knees if available\n",
        "        pts = []\n",
        "        for key in (\"l_hp\", \"r_hp\", \"l_kn\", \"r_kn\"):\n",
        "            p = get(KP[key])\n",
        "            if p is not None:\n",
        "                pts.append(p)\n",
        "\n",
        "        com = np.mean(pts, axis=0) if len(pts) else None\n",
        "        cog = com  # same approximation here\n",
        "\n",
        "        if com is not None:\n",
        "            com_xy = tuple(np.round(com).astype(int))\n",
        "            cv2.circle(frame, com_xy, 6, (255, 0, 255), -1)\n",
        "            cv2.putText(\n",
        "                frame,\n",
        "                f\"COM/CoG {person_id}\",\n",
        "                com_xy,\n",
        "                cv2.FONT_HERSHEY_SIMPLEX,\n",
        "                0.5,\n",
        "                (255, 0, 255),\n",
        "                1,\n",
        "                cv2.LINE_AA\n",
        "            )\n",
        "\n",
        "        la = get(KP[\"l_an\"])\n",
        "        ra = get(KP[\"r_an\"])\n",
        "\n",
        "        # Support line between ankles\n",
        "        if la is not None and ra is not None:\n",
        "            self.draw.line(frame, la, ra, (0, 200, 255), 2)\n",
        "\n",
        "        left_force = None\n",
        "        right_force = None\n",
        "        balance_vector = None\n",
        "        support_center = None\n",
        "\n",
        "        # Left leg force vector\n",
        "        if get(KP[\"l_kn\"]) is not None and la is not None and get(KP[\"l_hp\"]) is not None and com is not None:\n",
        "            v, m = self.force.compute(get(KP[\"l_kn\"]), la, get(KP[\"l_hp\"]), com)\n",
        "            self.draw.arrow(frame, la, v, (255, 0, 0), f\"{int(round(m))}\", scale=0.15)\n",
        "            left_force = {\"vector\": [float(v[0]), float(v[1])], \"mag\": float(m)}\n",
        "\n",
        "        # Right leg force vector\n",
        "        if get(KP[\"r_kn\"]) is not None and ra is not None and get(KP[\"r_hp\"]) is not None and com is not None:\n",
        "            v, m = self.force.compute(get(KP[\"r_kn\"]), ra, get(KP[\"r_hp\"]), com)\n",
        "            self.draw.arrow(frame, ra, v, (255, 0, 0), f\"{int(round(m))}\", scale=0.15)\n",
        "            right_force = {\"vector\": [float(v[0]), float(v[1])], \"mag\": float(m)}\n",
        "\n",
        "        # Stability\n",
        "        stable, score, support_center = self.stab.compute(com, la, ra, frame.shape[1])\n",
        "\n",
        "        # Balance vector: from COM to support center\n",
        "        if com is not None and support_center is not None:\n",
        "            balance_vector = support_center - com\n",
        "            self.draw.arrow(frame, com, balance_vector, (0, 255, 0) if stable else (0, 0, 255), \"Balance\", scale=1.0)\n",
        "\n",
        "        # Stability label on video\n",
        "        label = \"STABLE\" if stable else \"UNSTABLE\"\n",
        "        label_color = (0, 255, 0) if stable else (0, 0, 255)\n",
        "\n",
        "        label_pos = (30, 40 + 28 * person_id)\n",
        "        if com is not None:\n",
        "            label_pos = tuple(np.round(com + np.array([10, -10], dtype=np.float32)).astype(int))\n",
        "\n",
        "        cv2.putText(\n",
        "            frame,\n",
        "            f\"{label} ({score:.2f})\",\n",
        "            label_pos,\n",
        "            cv2.FONT_HERSHEY_SIMPLEX,\n",
        "            0.6,\n",
        "            label_color,\n",
        "            2,\n",
        "            cv2.LINE_AA\n",
        "        )\n",
        "\n",
        "        return {\n",
        "            \"person_id\": int(person_id),\n",
        "            \"com\": [float(com[0]), float(com[1])] if com is not None else None,\n",
        "            \"cog\": [float(cog[0]), float(cog[1])] if cog is not None else None,\n",
        "            \"support_center\": [float(support_center[0]), float(support_center[1])] if support_center is not None else None,\n",
        "            \"left_ankle\": [float(la[0]), float(la[1])] if la is not None else None,\n",
        "            \"right_ankle\": [float(ra[0]), float(ra[1])] if ra is not None else None,\n",
        "            \"left_force\": left_force,\n",
        "            \"right_force\": right_force,\n",
        "            \"balance_vector\": [float(balance_vector[0]), float(balance_vector[1])] if balance_vector is not None else None,\n",
        "            \"stability\": {\"stable\": bool(stable), \"score\": float(score)},\n",
        "            \"angles\": angles\n",
        "        }\n",
        "\n",
        "    def process_frame(self, frame):\n",
        "        results = self.model(frame, verbose=False)\n",
        "\n",
        "        frame_records = []\n",
        "        if not results or len(results) == 0 or results[0].keypoints is None:\n",
        "            cv2.putText(\n",
        "                frame,\n",
        "                \"No person detected\",\n",
        "                (30, 40),\n",
        "                cv2.FONT_HERSHEY_SIMPLEX,\n",
        "                1,\n",
        "                (0, 0, 255),\n",
        "                2,\n",
        "                cv2.LINE_AA\n",
        "            )\n",
        "            return frame, frame_records\n",
        "\n",
        "        r = results[0]\n",
        "\n",
        "        # Multi-person pose:\n",
        "        # keypoints.xy -> (N, 17, 2)\n",
        "        # keypoints.conf -> (N, 17)\n",
        "        kpts_all = r.keypoints.xy.cpu().numpy()\n",
        "        conf_all = r.keypoints.conf.cpu().numpy()\n",
        "\n",
        "        if len(kpts_all) == 0:\n",
        "            cv2.putText(\n",
        "                frame,\n",
        "                \"No person detected\",\n",
        "                (30, 40),\n",
        "                cv2.FONT_HERSHEY_SIMPLEX,\n",
        "                1,\n",
        "                (0, 0, 255),\n",
        "                2,\n",
        "                cv2.LINE_AA\n",
        "            )\n",
        "            return frame, frame_records\n",
        "\n",
        "        for pid, (kpts, conf) in enumerate(zip(kpts_all, conf_all)):\n",
        "            record = self.process_person(frame, kpts, conf, pid)\n",
        "            frame_records.append(record)\n",
        "\n",
        "            # Person label at nose if available, otherwise first valid point\n",
        "            if conf[KP[\"nose\"]] > MIN_CONF:\n",
        "                name_xy = tuple(np.round(kpts[KP[\"nose\"]]).astype(int))\n",
        "            else:\n",
        "                valid = kpts[conf > MIN_CONF]\n",
        "                if len(valid) > 0:\n",
        "                    name_xy = tuple(np.round(valid[0]).astype(int))\n",
        "                else:\n",
        "                    name_xy = (30, 80 + 25 * pid)\n",
        "\n",
        "            cv2.putText(\n",
        "                frame,\n",
        "                f\"Person {pid}\",\n",
        "                name_xy,\n",
        "                cv2.FONT_HERSHEY_SIMPLEX,\n",
        "                0.6,\n",
        "                (0, 255, 255),\n",
        "                2,\n",
        "                cv2.LINE_AA\n",
        "            )\n",
        "\n",
        "        return frame, frame_records\n",
        "\n",
        "# ---------------------------\n",
        "# RUN\n",
        "# ---------------------------\n",
        "cap = cv2.VideoCapture(INPUT_VIDEO)\n",
        "if not cap.isOpened():\n",
        "    raise FileNotFoundError(f\"Could not open input video: {INPUT_VIDEO}\")\n",
        "\n",
        "w = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))\n",
        "h = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))\n",
        "fps = cap.get(cv2.CAP_PROP_FPS)\n",
        "if fps is None or fps <= 1e-6:\n",
        "    fps = 30.0\n",
        "\n",
        "out = cv2.VideoWriter(\n",
        "    OUTPUT_VIDEO,\n",
        "    cv2.VideoWriter_fourcc(*\"mp4v\"),\n",
        "    fps,\n",
        "    (w, h)\n",
        ")\n",
        "\n",
        "engine = Engine()\n",
        "timeline = []\n",
        "frame_idx = 0\n",
        "\n",
        "while True:\n",
        "    ret, frame = cap.read()\n",
        "    if not ret:\n",
        "        break\n",
        "\n",
        "    annotated, people = engine.process_frame(frame)\n",
        "    out.write(annotated)\n",
        "\n",
        "    timeline.append({\n",
        "        \"frame\": int(frame_idx),\n",
        "        \"time\": round(frame_idx / fps, 4),\n",
        "        \"num_persons\": int(len(people)),\n",
        "        \"persons\": people\n",
        "    })\n",
        "\n",
        "    frame_idx += 1\n",
        "\n",
        "cap.release()\n",
        "out.release()\n",
        "\n",
        "with open(OUTPUT_JSON, \"w\") as f:\n",
        "    json.dump(timeline, f, indent=2)\n",
        "\n",
        "print(f\"Done. Saved video: {OUTPUT_VIDEO}\")\n",
        "print(f\"Done. Saved JSON: {OUTPUT_JSON}\")"
