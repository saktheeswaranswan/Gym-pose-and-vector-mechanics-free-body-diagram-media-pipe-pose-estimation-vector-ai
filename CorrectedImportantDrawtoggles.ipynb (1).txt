        "# ---------------------------\n",
        "# VECTOR UTILS\n",
        "# ---------------------------\n",
        "class Vec:\n",
        "\n",
        "    @staticmethod\n",
        "    def unit(v):\n",
        "        n = np.linalg.norm(v)\n",
        "        return v/n if n > 1e-6 else np.zeros_like(v)\n",
        "\n",
        "    @staticmethod\n",
        "    def angle(a, b, c):\n",
        "        ba = Vec.unit(a - b)\n",
        "        bc = Vec.unit(c - b)\n",
        "        return np.degrees(np.arccos(np.clip(np.dot(ba, bc), -1, 1)))\n",
        "\n",
        "# ---------------------------\n",
        "# DRAW CLASS (FIXED)\n",
        "# ---------------------------\n",
        "class Draw:\n",
        "\n",
        "    @staticmethod\n",
        "    def point(img, p, c=(0,255,255), r=5):\n",
        "        if p is None: return\n",
        "        cv2.circle(img, (int(p[0]), int(p[1])), r, c, -1)\n",
        "\n",
        "    @staticmethod\n",
        "    def line(img, a, b, c=(200,200,200), t=2):\n",
        "        if a is None or b is None: return\n",
        "        cv2.line(img, (int(a[0]),int(a[1])), (int(b[0]),int(b[1])), c, t)\n",
        "\n",
        "    @staticmethod\n",
        "    def text(img, t, p, c=(255,255,255), s=0.5):\n",
        "        if p is None: return\n",
        "        cv2.putText(img, t, (int(p[0]),int(p[1])),\n",
        "                    cv2.FONT_HERSHEY_SIMPLEX, s, c, 1)\n",
        "\n",
        "    # ✅ FIXED ARC FUNCTION\n",
        "    @staticmethod\n",
        "    def arc(img, a, b, c, color=(0,255,255), radius=25):\n",
        "\n",
        "        a = np.array(a); b = np.array(b); c = np.array(c)\n",
        "\n",
        "        ba = a - b\n",
        "        bc = c - b\n",
        "\n",
        "        if np.linalg.norm(ba) < 1e-6 or np.linalg.norm(bc) < 1e-6:\n",
        "            return\n",
        "\n",
        "        ang1 = math.atan2(ba[1], ba[0])\n",
        "        ang2 = math.atan2(bc[1], bc[0])\n",
        "\n",
        "        if ang2 < ang1:\n",
        "            ang1, ang2 = ang2, ang1\n",
        "\n",
        "        for t in np.linspace(ang1, ang2, 20):\n",
        "            x = int(b[0] + radius * math.cos(t))\n",
        "            y = int(b[1] + radius * math.sin(t))\n",
        "            cv2.circle(img, (x,y), 1, color, -1)\n",
        "\n",
        "# ---------------------------\n",
        "# ENGINE\n",
        "# ---------------------------\n",
        "class Engine:\n",
        "\n",
        "    def __init__(self):\n",
        "        self.model = YOLO(MODEL_PATH)\n",
        "        self.draw = Draw()\n",
        "\n",
        "    def process_person(self, frame, kpts, conf):\n",
        "\n",
        "        # draw skeleton\n",
        "        for a,b in SKELETON:\n",
        "            if conf[a] > MIN_CONF and conf[b] > MIN_CONF:\n",
        "                self.draw.line(frame, kpts[a], kpts[b])\n",
        "\n",
        "        # draw joints\n",
        "        for name,(a,b,c) in JOINTS.items():\n",
        "            if conf[a]>MIN_CONF and conf[b]>MIN_CONF and conf[c]>MIN_CONF:\n",
        "\n",
        "                pa, pb, pc = kpts[a], kpts[b], kpts[c]\n",
        "                ang = Vec.angle(pa,pb,pc)\n",
        "\n",
        "                if DRAW_JOINT_ANGLES:\n",
        "                    self.draw.arc(frame, pa,pb,pc)\n",
        "                    self.draw.text(frame, f\"{int(ang)}°\",\n",
        "                                   (pb[0]+5, pb[1]-5))\n",
        "\n",
        "        return frame\n",
        "\n",
        "    def process_frame(self, frame):\n",
        "\n",
        "        results = self.model(frame, verbose=False)\n",
        "\n",
        "        if not results or results[0].keypoints is None:\n",
        "            return frame\n",
        "\n",
        "        kpts_all = results[0].keypoints.xy.cpu().numpy()\n",
        "        conf_all = results[0].keypoints.conf.cpu().numpy()\n",
        "\n",
        "        for kpts, conf in zip(kpts_all, conf_all):\n",
        "            frame = self.process_person(frame, kpts, conf)\n",
        "\n",
        "        return frame\n",
