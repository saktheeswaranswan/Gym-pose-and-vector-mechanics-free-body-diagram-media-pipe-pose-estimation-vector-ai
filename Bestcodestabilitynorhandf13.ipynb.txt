        "# ---------------------------\n",
        "# INSTALL\n",
        "# ---------------------------\n",
        "!pip install ultralytics opencv-python numpy -q\n",
        "\n",
        "import cv2\n",
        "import numpy as np\n",
        "import math\n",
        "from ultralytics import YOLO\n",
        "from collections import deque\n",
        "\n",
        "# ---------------------------\n",
        "# CONFIG\n",
        "# ---------------------------\n",
        "INPUT_VIDEO = \"video.mp4\"\n",
        "OUTPUT_VIDEO = \"inference_video.mp4\"\n",
        "MODEL_PATH = \"yolo11n-pose.pt\"\n",
        "\n",
        "MIN_CONF = 0.3\n",
        "HISTORY = 10\n",
        "\n",
        "# ---------------------------\n",
        "# KEYPOINTS\n",
        "# ---------------------------\n",
        "KP = {\n",
        "    \"l_sh\": 5, \"r_sh\": 6,\n",
        "    \"l_el\": 7, \"r_el\": 8,\n",
        "    \"l_wr\": 9, \"r_wr\": 10,\n",
        "    \"l_hp\": 11, \"r_hp\": 12,\n",
        "    \"l_kn\": 13, \"r_kn\": 14,\n",
        "    \"l_an\": 15, \"r_an\": 16\n",
        "}\n",
        "\n",
        "# ---------------------------\n",
        "# VECTOR UTIL\n",
        "# ---------------------------\n",
        "class Vec:\n",
        "    @staticmethod\n",
        "    def unit(v):\n",
        "        n = np.linalg.norm(v)\n",
        "        return v / n if n > 1e-6 else np.zeros_like(v)\n",
        "\n",
        "    @staticmethod\n",
        "    def dot(a, b):\n",
        "        return float(np.dot(a, b))\n",
        "\n",
        "    @staticmethod\n",
        "    def cross2d(a, b):\n",
        "        return float(a[0]*b[1] - a[1]*b[0])\n",
        "\n",
        "# ---------------------------\n",
        "# BIOMECHANICS\n",
        "# ---------------------------\n",
        "class Biomech:\n",
        "\n",
        "    @staticmethod\n",
        "    def compute_com(kpts, conf):\n",
        "        pts = [np.array(p) for i,p in enumerate(kpts) if conf[i]>MIN_CONF]\n",
        "        return np.mean(pts, axis=0) if len(pts)>0 else None\n",
        "\n",
        "    @staticmethod\n",
        "    def compute_cop(l_an, r_an, com):\n",
        "        if l_an is None and r_an is None:\n",
        "            return None\n",
        "        if l_an is None:\n",
        "            return r_an\n",
        "        if r_an is None:\n",
        "            return l_an\n",
        "        if com is None:\n",
        "            return (l_an + r_an) / 2\n",
        "\n",
        "        dl = abs(com[0] - l_an[0])\n",
        "        dr = abs(com[0] - r_an[0])\n",
        "\n",
        "        wl = 1 / (dl + 1e-6)\n",
        "        wr = 1 / (dr + 1e-6)\n",
        "\n",
        "        s = wl + wr\n",
        "        wl /= s\n",
        "        wr /= s\n",
        "\n",
        "        return wl * l_an + wr * r_an\n",
        "\n",
        "    @staticmethod\n",
        "    def stability(com, cop):\n",
        "        if com is None or cop is None:\n",
        "            return False, 0.0\n",
        "\n",
        "        dx = abs(com[0] - cop[0])\n",
        "        score = math.exp(-dx / 50)\n",
        "        return score > 0.55, score\n",
        "\n",
        "    @staticmethod\n",
        "    def joint_torque(joint, com, force):\n",
        "        if joint is None or com is None:\n",
        "            return 0.0\n",
        "        r = com - joint\n",
        "        return Vec.cross2d(r, force)\n",
        "\n",
        "    @staticmethod\n",
        "    def leg_normal_force(hip, ankle, com):\n",
        "        if hip is None or ankle is None or com is None:\n",
        "            return np.zeros(2, dtype=np.float32)\n",
        "\n",
        "        leg = Vec.unit(ankle - hip)\n",
        "        gravity = np.array([0, 1], dtype=np.float32)\n",
        "\n",
        "        load = Vec.dot(gravity, leg)\n",
        "        return leg * load * 150\n",
        "\n",
        "    @staticmethod\n",
        "    def hand_force(sh, el, wr):\n",
        "        if sh is None or el is None or wr is None:\n",
        "            return np.zeros(2, dtype=np.float32)\n",
        "\n",
        "        arm = Vec.unit(wr - sh)\n",
        "        return arm * 80\n",
        "\n",
        "\n",
        "# ---------------------------\n",
        "# DRAW\n",
        "# ---------------------------\n",
        "class Draw:\n",
        "\n",
        "    @staticmethod\n",
        "    def point(img, p, c, r=5):\n",
        "        cv2.circle(img, (int(p[0]), int(p[1])), r, c, -1)\n",
        "\n",
        "    @staticmethod\n",
        "    def arrow(img, p, v, scale=0.4, c=(0,0,255)):\n",
        "        end = p + v * scale\n",
        "        cv2.arrowedLine(img,\n",
        "                        (int(p[0]), int(p[1])),\n",
        "                        (int(end[0]), int(end[1])),\n",
        "                        c, 2)\n",
        "\n",
        "    @staticmethod\n",
        "    def text(img, t, p, c=(255,255,255)):\n",
        "        cv2.putText(img, t,\n",
        "                    (int(p[0]), int(p[1])),\n",
        "                    cv2.FONT_HERSHEY_SIMPLEX,\n",
        "                    0.5, c, 1)\n",
        "\n",
        "\n",
        "# ---------------------------\n",
        "# ENGINE\n",
        "# ---------------------------\n",
        "class Engine:\n",
        "\n",
        "    def __init__(self):\n",
        "        self.model = YOLO(MODEL_PATH)\n",
        "        self.history = {}\n",
        "\n",
        "    def predict_fall(self, pid, com):\n",
        "        if com is None:\n",
        "            return False\n",
        "\n",
        "        if pid not in self.history:\n",
        "            self.history[pid] = deque(maxlen=HISTORY)\n",
        "\n",
        "        hist = self.history[pid]\n",
        "        hist.append(com.copy())\n",
        "\n",
        "        if len(hist) < HISTORY:\n",
        "            return False\n",
        "\n",
        "        velocities = [hist[i+1][0] - hist[i][0] for i in range(len(hist)-1)]\n",
        "        trend = np.mean(velocities)\n",
        "\n",
        "        return abs(trend) > 5\n",
        "\n",
        "    def process(self, frame):\n",
        "\n",
        "        results = self.model(frame, verbose=False)\n",
        "        if results[0].keypoints is None:\n",
        "            return frame\n",
        "\n",
        "        kpts_all = results[0].keypoints.xy.cpu().numpy()\n",
        "        conf_all = results[0].keypoints.conf.cpu().numpy()\n",
        "\n",
        "        for pid, (kpts, conf) in enumerate(zip(kpts_all, conf_all)):\n",
        "\n",
        "            pts = {k: np.array(kpts[i]) if conf[i]>MIN_CONF else None\n",
        "                   for k,i in KP.items()}\n",
        "\n",
        "            com = Biomech.compute_com(kpts, conf)\n",
        "            cop = Biomech.compute_cop(pts[\"l_an\"], pts[\"r_an\"], com)\n",
        "\n",
        "            stable, score = Biomech.stability(com, cop)\n",
        "            fall = self.predict_fall(pid, com)\n",
        "\n",
        "            grf = np.array([0, -150], dtype=np.float32)\n",
        "\n",
        "            torque_l = Biomech.joint_torque(pts[\"l_kn\"], com, grf)\n",
        "            torque_r = Biomech.joint_torque(pts[\"r_kn\"], com, grf)\n",
        "\n",
        "            nf_l = Biomech.leg_normal_force(pts[\"l_hp\"], pts[\"l_an\"], com)\n",
        "            nf_r = Biomech.leg_normal_force(pts[\"r_hp\"], pts[\"r_an\"], com)\n",
        "\n",
        "            hf_l = Biomech.hand_force(pts[\"l_sh\"], pts[\"l_el\"], pts[\"l_wr\"])\n",
        "            hf_r = Biomech.hand_force(pts[\"r_sh\"], pts[\"r_el\"], pts[\"r_wr\"])\n",
        "\n",
        "            # DRAW\n",
        "            if com is not None:\n",
        "                Draw.point(frame, com, (255,0,255), 7)\n",
        "\n",
        "            if cop is not None:\n",
        "                Draw.point(frame, cop, (0,255,0), 7)\n",
        "                Draw.arrow(frame, cop, grf, c=(255,255,0))\n",
        "\n",
        "            if pts[\"l_kn\"] is not None:\n",
        "                Draw.arrow(frame, pts[\"l_kn\"], nf_l, c=(0,255,255))\n",
        "                Draw.text(frame, f\"T:{torque_l:.1f}\", pts[\"l_kn\"]+[0,20])\n",
        "\n",
        "            if pts[\"r_kn\"] is not None:\n",
        "                Draw.arrow(frame, pts[\"r_kn\"], nf_r, c=(0,255,255))\n",
        "                Draw.text(frame, f\"T:{torque_r:.1f}\", pts[\"r_kn\"]+[0,20])\n",
        "\n",
        "            if pts[\"l_wr\"] is not None:\n",
        "                Draw.arrow(frame, pts[\"l_wr\"], hf_l, c=(255,0,0))\n",
        "\n",
        "            if pts[\"r_wr\"] is not None:\n",
        "                Draw.arrow(frame, pts[\"r_wr\"], hf_r, c=(255,0,0))\n",
        "\n",
        "            Draw.text(frame,\n",
        "                      f\"P{pid} {'STABLE' if stable else 'UNSTABLE'} {score:.2f}\",\n",
        "                      (30,40+pid*20),\n",
        "                      (0,255,0) if stable else (0,0,255))\n",
        "\n",
        "            if fall:\n",
        "                Draw.text(frame, \"FALL RISK\", (200,40+pid*20), (0,0,255))\n",
        "\n",
        "        return frame\n",
        "\n",
        "\n",
        "# ---------------------------\n",
        "# MAIN\n",
        "# ---------------------------\n",
        "def main():\n",
        "    cap = cv2.VideoCapture(INPUT_VIDEO)\n",
        "\n",
        "    w = int(cap.get(3))\n",
        "    h = int(cap.get(4))\n",
        "    fps = cap.get(5) or 30\n",
        "\n",
        "    out = cv2.VideoWriter(\n",
        "        OUTPUT_VIDEO,\n",
        "        cv2.VideoWriter_fourcc(*\"mp4v\"),\n",
        "        fps, (w,h)\n",
        "    )\n",
        "\n",
        "    engine = Engine()\n",
        "\n",
        "    while True:\n",
        "        ret, frame = cap.read()\n",
        "        if not ret:\n",
        "            break\n",
        "\n",
        "        frame = engine.process(frame)\n",
        "        out.write(frame)\n",
        "\n",
        "    cap.release()\n",
        "    out.release()\n",
        "\n",
        "    print(\"✅ Done:\", OUTPUT_VIDEO)\n",
        "\n",
        "\n",
        "if __name__ == \"__main__\":\n",
        "    main()"
